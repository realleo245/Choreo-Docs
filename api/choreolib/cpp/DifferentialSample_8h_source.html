106 constexpr DifferentialSample Interpolate(const DifferentialSample& endValue,
107 units::second_t t) const {
108 units::scalar_t scale = (t - timestamp) / (endValue.timestamp - timestamp);
109 frc::Pose2d interpolatedPose =
110 frc::Interpolate(GetPose(), endValue.GetPose(), scale.value());
111
112 return DifferentialSample{
113 wpi::Lerp(timestamp, endValue.timestamp, scale),
114 interpolatedPose.X(),
115 interpolatedPose.Y(),
116 interpolatedPose.Rotation().Radians(),
117 wpi::Lerp(vl, endValue.vl, scale),
118 wpi::Lerp(vr, endValue.vr, scale),
119 wpi::Lerp(al, endValue.al, scale),
120 wpi::Lerp(ar, endValue.ar, scale),
121 wpi::Lerp(fl, endValue.fl, scale),
122 wpi::Lerp(fr, endValue.fr, scale),
123 };
124 }

142 fr, fl);

152 constexpr bool operator==(const DifferentialSample& other) const {
153 constexpr double epsilon = 1e-6;
154
155 auto compare_units = [epsilon](const auto& a, const auto& b) {
156 using UnitType =
157 std::remove_const_t<std::remove_reference_t<decltype(a)>>;
158 return units::math::abs(a - b) < UnitType(epsilon);
159 };
160
161 return compare_units(timestamp, other.timestamp) &&
162 compare_units(x, other.x) && compare_units(y, other.y) &&
163 compare_units(heading, other.heading) &&
164 compare_units(vl, other.vl) && compare_units(vr, other.vr) &&
165 compare_units(al, other.al) && compare_units(ar, other.ar) &&
166 compare_units(fl, other.fl) && compare_units(fr, other.fr);
167 }
